function renderDonut(){
  const total = factors.length;
  let acc = 0;
  const stops = [];
  factors.forEach(f => {
    const deg = 360 / total;
    stops.push(`${f.color} ${acc.toFixed(1)}deg ${(acc+deg).toFixed(1)}deg`);
    acc += deg;
  });
  document.getElementById('donutEl').style.background = `conic-gradient(${stops.join(',')})`;
  document.getElementById('donutN').textContent = total;
  document.getElementById('legendEl').innerHTML = factors.map(f =>
    `<li><span><span class="sw" style="background:${f.color}"></span>${f.name}</span><b>${f.impact}%</b></li>`
  ).join('');
}

}

function renderComparison(){

  let html = '<thead><tr>';
  headers.forEach(h => html += `<th>${h}</th>`);
  html += '</tr></thead><tbody>';
  
  factors.forEach((f, i) => {
    html += `<tr>
      <td><span class="factor-name" style="color:${f.color};">${f.name}</span></td>
      <td>${types[i]}</td>
      <td><span class="factor-score">${f.impact}%</span></td>
      <td>${f.populationSize}</td>

  });
  
  html += '</tbody>';
  document.getElementById('comparisonTable').innerHTML = html;
}

// Page nav active state on scroll
const screens = document.querySelectorAll('.screen');
const navObserver = new IntersectionObserver((entries) => {
  entries.forEach(en => {
    if(en.isIntersecting){
      document.querySelectorAll('.pagenav a').forEach(l => 
        l.classList.toggle('active', l.dataset.target === en.target.id)
      );
    }
  });
}, {threshold:0.5});
screens.forEach(s => navObserver.observe(s));

// Collapse functionality
document.querySelectorAll('.collapse-btn[data-target]').forEach(btn => {
  btn.addEventListener('click', () => {
    const target = document.getElementById(btn.dataset.target);
    const isCollapsed = target.classList.toggle('collapsed');
    btn.textContent = isCollapsed ? '⌃ Mở rộng' : '⌄ Thu gọn';
  });
});

const collapseAllBtn = document.getElementById('collapseAllBtn');
collapseAllBtn.addEventListener('click', () => {
  const shouldCollapse = collapseAllBtn.textContent.includes('Thu gọn');
  document.querySelectorAll('.collapse-btn[data-target]').forEach(btn => {
    const target = document.getElementById(btn.dataset.target);
    target.classList.toggle('collapsed', shouldCollapse);
    btn.textContent = shouldCollapse ? '⌃ Mở rộng' : '⌄ Thu gọn';
  });
  collapseAllBtn.textContent = shouldCollapse ? '⌃ Mở rộng tất cả' : '⌄ Thu gọn tất cả';
});
